12void ei_impl_placer_run(ei_widget_t widget) {
13 if (widget == NULL || widget->placer_params == NULL || widget->parent == NULL) {
14 // Ne pas placer le root_widget via le placeur, ou si pas de parent/params
15 if (widget == ei_app_root_widget() && widget->parent == NULL) return;
16 // fprintf(stderr, "ei_impl_placer_run: widget, placer_params ou parent est NULL.\n");
17 return;
18 }
19
20 ei_impl_placer_params_t* params = widget->placer_params;
21 const ei_rect_t* parent_content_rect = ei_widget_get_content_rect(widget->parent);
22
23 if (parent_content_rect == NULL) {
24 fprintf(stderr, "ei_impl_placer_run: Parent de %p n'a pas de content_rect.\n", (void*)widget);
25 return;
26 }
27
28 // 1. Calcul de la taille finale du widget (width, height)
29 ei_size_t final_size;
30
31 // Largeur
32 if (params->width >= 0) { // Priorité à la largeur absolue
33 final_size.width = params->width;
34 } else if (params->rel_width >= 0.0f) { // Puis relative
35 final_size.width = (int)roundf(params->rel_width * parent_content_rect->size.width);
36 } else { // Sinon, taille demandée par le widget
37 final_size.width = widget->requested_size.width;
38 }
39
40 // Hauteur
41 if (params->height >= 0) { // Priorité à la hauteur absolue
42 final_size.height = params->height;
43 } else if (params->rel_height >= 0.0f) { // Puis relative
44 final_size.height = (int)roundf(params->rel_height * parent_content_rect->size.height);
45 } else { // Sinon, taille demandée par le widget
46 final_size.height = widget->requested_size.height;
47 }
48
49 // 2. Calcul de la position du point d'ancrage du widget DANS le parent
50 ei_point_t anchor_pos_in_parent;
51 anchor_pos_in_parent.x = params->x + (int)roundf(params->rel_x * parent_content_rect->size.width);
52 anchor_pos_in_parent.y = params->y + (int)roundf(params->rel_y * parent_content_rect->size.height);
53
54 // Le point d'ancrage est relatif au coin top-left du content_rect du parent
55 anchor_pos_in_parent.x += parent_content_rect->top_left.x;
56 anchor_pos_in_parent.y += parent_content_rect->top_left.y;
57
58 // 3. Calcul de la position top-left réelle du widget sur l'écran, en fonction de l'ancre
59 widget->screen_location.top_left = anchor_pos_in_parent; // Initialisation pour anc_northwest
60
61 switch (params->anchor) {
62 case ei_anc_northwest:
63 // Déjà correct
64 break;
65 case ei_anc_north:
66 widget->screen_location.top_left.x = anchor_pos_in_parent.x - final_size.width / 2;
67 break;
68 case ei_anc_northeast:
69 widget->screen_location.top_left.x = anchor_pos_in_parent.x - final_size.width;
70 break;
71 case ei_anc_west:
72 widget->screen_location.top_left.y = anchor_pos_in_parent.y - final_size.height / 2;
73 break;
74 case ei_anc_center:
75 widget->screen_location.top_left.x = anchor_pos_in_parent.x - final_size.width / 2;
76 widget->screen_location.top_left.y = anchor_pos_in_parent.y - final_size.height / 2;
77 break;
78 case ei_anc_east:
79 widget->screen_location.top_left.x = anchor_pos_in_parent.x - final_size.width;
80 widget->screen_location.top_left.y = anchor_pos_in_parent.y - final_size.height / 2;
81 break;
82 case ei_anc_southwest:
83 widget->screen_location.top_left.y = anchor_pos_in_parent.y - final_size.height;
84 break;
85 case ei_anc_south:
86 widget->screen_location.top_left.x = anchor_pos_in_parent.x - final_size.width / 2;
87 widget->screen_location.top_left.y = anchor_pos_in_parent.y - final_size.height;
88 break;
89 case ei_anc_southeast:
90 widget->screen_location.top_left.x = anchor_pos_in_parent.x - final_size.width;
91 widget->screen_location.top_left.y = anchor_pos_in_parent.y - final_size.height;
92 break;
93 case ei_anc_none: // Devrait être traité comme northwest par défaut selon la doc
94 default:
95 // ei_anc_northwest est déjà le comportement par défaut
96 break;
97 }
98 widget->screen_location.size = final_size;
99
100 // 4. Mettre à jour le content_rect du widget
101 // Le content_rect est screen_location moins les bordures.
102 // Si le widget n'a pas de content_rect distinct alloué, on ne peut pas le modifier directement
103 // s'il pointe vers screen_location. Il faut que les classes gèrent cela.
104 // Pour l'instant, on laisse content_rect pointer vers screen_location et les classes
105 // s'occuperont de calculer la zone de contenu pour leurs enfants.
106 // Une meilleure solution serait que ei_impl_widget_t ait un champ ei_rect_t content_rect_value
107 // et que content_rect pointe dessus, puis on le met à jour ici.
108 // Pour la version A.5, content_rect = screen_location est acceptable.
109 // Pour une version plus avancée:
110 if (widget->content_rect == &(widget->screen_location)) {
111 // Si content_rect pointe vers screen_location, il est mis à jour automatiquement.
112 // Cependant, pour avoir un VRAI content_rect (intérieur aux bordures), il faudrait
113 // soit une copie, soit une structure dédiée.
114 // On va supposer que les drawfunc des widgets calculent la zone pour leurs enfants
115 // à partir de screen_location et border_width.
116 } else if (widget->content_rect != NULL) { // Si un content_rect distinct est alloué
117 widget->content_rect->top_left.x = widget->screen_location.top_left.x + widget->border_width;
118 widget->content_rect->top_left.y = widget->screen_location.top_left.y + widget->border_width;
119 widget->content_rect->size.width = widget->screen_location.size.width - 2 * widget->border_width;
120 widget->content_rect->size.height = widget->screen_location.size.height - 2 * widget->border_width;
121 if (widget->content_rect->size.width < 0) widget->content_rect->size.width = 0;
122 if (widget->content_rect->size.height < 0) widget->content_rect->size.height = 0;
123 }
124
125
126 // Notifier le widget que sa géométrie a changé
127 if (widget->wclass && widget->wclass->geomnotifyfunc) {
128 widget->wclass->geomnotifyfunc(widget);
129 }
130
131 // Mettre à jour la géométrie des enfants récursivement
132 ei_widget_t child = widget->children_head;
133 while (child != NULL) {
134 ei_impl_placer_run(child);
135 child = child->next_sibling;
136 }
137 // Invalider la nouvelle zone pour redessin (sera fait par l'application)
138 ei_app_invalidate_rect(&(widget->screen_location));
139}

142void ei_place(ei_widget_t widget,
143 ei_anchor_t* anchor,
144 int* x, int* y,
145 int* width, int* height,
146 float* rel_x,
147 float* rel_y,
148 float* rel_width,
149 float* rel_height)
150{
151 if (widget == NULL) return;
152
153 ei_rect_t old_location = widget->screen_location; // Sauvegarder l'ancienne position
154
155 if (widget->placer_params == NULL) {
156 widget->placer_params = malloc(sizeof(ei_impl_placer_params_t));
157 if (widget->placer_params == NULL) {
158 fprintf(stderr, "ei_place: Erreur d'allocation pour placer_params\n");
159 return;
160 }
161 // Valeurs par défaut spécifiées dans la documentation de ei_place
162 widget->placer_params->anchor = ei_anc_northwest;
163 widget->placer_params->x = 0;
164 widget->placer_params->y = 0;
165 widget->placer_params->width = -1; // Signifie: utiliser requested_size ou rel_width si défini
166 widget->placer_params->height = -1; // Signifie: utiliser requested_size ou rel_height si défini
167 widget->placer_params->rel_x = 0.0f;
168 widget->placer_params->rel_y = 0.0f;
169 widget->placer_params->rel_width = -1.0f; // Signifie: non utilisé ou utiliser width si défini
170 widget->placer_params->rel_height= -1.0f; // Signifie: non utilisé ou utiliser height si défini
171 }
172
173 ei_impl_placer_params_t* params = widget->placer_params;
174
175 if (anchor) params->anchor = *anchor;
176 if (x) params->x = *x;
177 if (y) params->y = *y;
178 if (width) params->width = *width; else if (width == NULL && params->width == -1) params->width = -1; // Conserver -1 si NULL
179 if (height) params->height = *height; else if (height == NULL && params->height == -1) params->height = -1;
180 if (rel_x) params->rel_x = *rel_x;
181 if (rel_y) params->rel_y = *rel_y;
182 if (rel_width) params->rel_width = *rel_width; else if (rel_width == NULL && params->rel_width == -1.0f) params->rel_width = -1.0f;
183 if (rel_height) params->rel_height = *rel_height; else if (rel_height == NULL && params->rel_height == -1.0f) params->rel_height = -1.0f;
184
185 // Si une dimension relative est fournie, la dimension absolue correspondante doit être ignorée
186 // (sauf si l'absolue est explicitement donnée et > -1, elle a priorité sur la rel_size dans mon impl_placer_run)
187 // La doc de ei_place dit: "If no size is provided (either absolute or relative), then either the requested size...".
188 // Ma logique dans impl_placer_run: abs > rel > requested.
189
190 if (widget->parent != NULL) { // Ne pas essayer de placer le root widget
191 ei_app_invalidate_rect(&old_location); // Invalider l'ancienne position
192 ei_impl_placer_run(widget); // Appliquer les nouveaux paramètres et invalider la nouvelle
193 }
194}
