7// Cette fonction remplit une zone avec une couleur (comme si on peignait un mur !)
8void ei_fill(ei_surface_t surface, const ei_color_t* couleur, const ei_rect_t* clipper)
9{
10 // On récupère le buffer (l'endroit où on dessine) et la taille de la surface
11 uint8_t* pixel_0 = hw_surface_get_buffer(surface);
12 ei_size_t taille_surface = hw_surface_get_size(surface);
13
14 // On transforme la couleur en un format que l'écran comprend (Mac/Windows ou Linux, c'est pas pareil)
15 #if defined(_APPLE_) || defined(_WIN32)
16 uint32_t valeur_pixel = ei_impl_map_rgba(surface, *couleur);
17 #else
18 uint32_t valeur_pixel = *((uint32_t*)&*couleur);
19 #endif
20
21 // On définit la zone qu’on va remplir (le clipper, c’est comme une fenêtre où on a le droit de peindre)
22 int clip_xmin = 0, clip_ymin = 0, clip_xmax = taille_surface.width - 1, clip_ymax = taille_surface.height - 1;
23 if (clipper != NULL) {
24 clip_xmin = clipper->top_left.x;
25 clip_ymin = clipper->top_left.y;
26 clip_xmax = clip_xmin + clipper->size.width - 1;
27 clip_ymax = clip_ymin + clipper->size.height - 1;
28 }
29
30 // On remplit juste la zone du clipper, ligne par ligne
31 for (int y = clip_ymin; y <= clip_ymax; y++) {
32 // On trouve le début de la ligne
33 uint8_t* ptr_ligne = pixel_0 + (y * taille_surface.width * 4) + (clip_xmin * 4);
34 uint32_t* ptr_pixel = (uint32_t*)ptr_ligne;
35 // On colore chaque pixel de la ligne
36 for (int x = clip_xmin; x <= clip_xmax; x++) {
37 *ptr_pixel++ = valeur_pixel;
38 }
39 }

42}

62// Dessine un polygone rempli (genre un octogone tout coloré !)
63void ei_draw_polygon(ei_surface_t surface, ei_point_t* points, size_t taille_points,
64 ei_color_t couleur, const ei_rect_t* clipper)
65{
66 // S’il n’y a pas assez de points ou c’est vide, on dégage
67 if (points == NULL || taille_points < 3) return;
68
69 // On récupère la taille de l’écran
70 ei_size_t taille_surface = hw_surface_get_size(surface);
71
72 // On cherche les y minimum et maximum du polygone
73 int y_min = points[0].y;
74 int y_max = points[0].y;
75 for (size_t i = 1; i < taille_points; i++) {
76 if (points[i].y < y_min) y_min = points[i].y;
77 if (points[i].y > y_max) y_max = points[i].y;
78 }
79
80 // On ajuste les limites pour pas dessiner n’importe où
81 int clip_ymin = y_min, clip_ymax = y_max;
82 if (clipper != NULL) {
83 // On prend le max/min entre le clipper et le polygone
84 clip_ymin = (clip_ymin > clipper->top_left.y) ? clip_ymin : clipper->top_left.y;
85 clip_ymax = (clip_ymax < (clipper->top_left.y + clipper->size.height)) ? clip_ymax : (clipper->top_left.y + clipper->size.height);
86 }
87 clip_ymin = (clip_ymin > 0) ? clip_ymin : 0;
88 clip_ymax = (clip_ymax < taille_surface.height) ? clip_ymax : (taille_surface.height);
89
90 // Si la zone est vide, on sort
91 if (clip_ymin > clip_ymax) return;
92
93 // On crée une table pour stocker les côtés (avec calloc, merci <stdlib.h> !)
94 edge_t** table_cotes = calloc(clip_ymax - clip_ymin, sizeof(edge_t));
95 // On remplit la table avec les arêtes du polygone
96 creer_table_tc(points, taille_points, table_cotes, clip_ymin, clip_ymax);
97
98 // On commence avec une liste vide pour les arêtes actives (TCA)
99 edge_t* tca = NULL;
100
101 // On boucle sur chaque ligne (scanline) de y_min à y_max
102 for (int y = clip_ymin; y <= clip_ymax; y++) {
103 // On vire les arêtes qui se terminent à cette ligne
104 supprimer_arete_tca(y, &tca);
105
106 // On ajoute les nouvelles arêtes qui commencent à cette ligne
107 if ((y - clip_ymin) >= 0 && (y - clip_ymin) < clip_ymax - clip_ymin + 1) {
108 edge_t* cote = table_cotes[y - clip_ymin];
109 while (cote != NULL) {
110 edge_t* next = cote->next;
111 ajoute_arete_tca(cote, &tca); // On met l’arête dans la TCA
112 cote = next;
113 }
114 table_cotes[y - clip_ymin] = NULL; // On vide cette case
115 }
116
117 // On trie la TCA pour avoir les arêtes dans l’ordre des x
118 trier_tca_x(&tca);
119
120 // On remplit les bouts entre les arêtes (ça fait le polygone !)
121 edge_t* courant = tca;
122 while (courant != NULL && courant->next != NULL) {
123 // On trouve les x de début et de fin pour cette ligne
124 int x_debut = (int)ceilf(courant->x);
125 int x_fin = (int)floorf(courant->next->x);
126
127 // S’il y a quelque chose à dessiner, on trace une ligne horizontale
128 if (x_debut <= x_fin) {
129 draw_horizontal_line(surface, x_debut, x_fin, y, couleur, clipper);
130 }
131 courant = courant->next->next; // On passe au prochain couple
132 }
133
134 // On met à jour les x pour la ligne suivante
135 courant = tca;
136 while (courant != NULL) {
137 courant->x += courant->inv_m;
138 courant = courant->next;
139 }
140 }
141
142 // On nettoie la TCA (on libère la mémoire avec free, merci <stdlib.h> !)
143 while (tca != NULL) {
144 edge_t* u = tca;
145 tca = tca->next;
146 free(u);
147 }
148 // On nettoie aussi la table des côtés
149 for (int i = 0; i < clip_ymax - clip_ymin + 1; i++) {
150 edge_t* cote = table_cotes[i];
151 while (cote != NULL) {
152 edge_t* u = cote;
153 cote = cote->next;
154 free(u);
155 }
156 }
157 // On libère la table elle-même
158 free(table_cotes);
159}

163 // Bon, d'abord on vérifie si y'a pas un truc qui manque, genre le texte ou les coordonnées
164 if (text == NULL || where == NULL) return ;
165 // Là on choisit la police : si y'en a pas, on prend la police par défaut, tranquille

167 // On crée une surface avec le texte, genre on dessine le texte tout seul dans un coin

169 // On fait un rectangle pour dire où on va coller le texte, avec sa taille et tout
170 ei_rect_t dst_rect = ei_rect(*where, hw_surface_get_size(surface_TXT));
171 // Là c'est le rectangle source, genre toute la surface du texte, depuis le coin en haut à gauche
172 ei_rect_t src_rect = ei_rect(ei_point_zero(), hw_surface_get_size(surface_TXT));
173 // On coupe tout ce qui dépasse si y'a un clipper
174 dst_rect = intersection_rect(&dst_rect, clipper, &src_rect);
175 // On copie le texte sur la surface finale, avec du flou (alpha true), ça fait classe
176 ei_copy_surface(surface, &dst_rect, surface_TXT, &src_rect, true);
177 // Faut pas oublier de libérer la surface temporaire, sinon c'est le bordel dans la mémoire
178 hw_surface_free(surface_TXT);

182int ei_copy_surface(ei_surface_t destination, const ei_rect_t* dst_rect, ei_surface_t source, const ei_rect_t* src_rect, bool alpha) {
183 // Oof, on check vite fait si les surfaces sont pas null
184 if (!destination || !source ) return 1 ;
185
186 ei_size_t dst_surface_size = hw_surface_get_size(destination);
187 ei_size_t src_surface_size = hw_surface_get_size(source);
188
189 // Si y’a pas de rectangle donné, bah on prend toute la surface, easy
190 ei_rect_t dst_rect_reel = dst_rect ? *dst_rect : ei_rect(ei_point_zero(), hw_surface_get_size(destination));
191 ei_rect_t src_rect_reel = src_rect ? *src_rect : ei_rect(ei_point_zero(), hw_surface_get_size(source));
192 // Faut que les rectangles aient la même taille
193 if (dst_rect_reel.size.width != src_rect_reel.size.width || dst_rect_reel.size.height != src_rect_reel.size.height) return 1 ;
194 // Si le rectangle est vide
195 if (dst_rect_reel.size.width <= 0 || dst_rect_reel.size.height <= 0) return 0 ;
196 // On récupère les adresses des pixels, genre le début des données
197 uint8_t* pixel_0_dst = hw_surface_get_buffer(destination);
198 uint8_t* pixel_0_src = hw_surface_get_buffer(source);
199 // Les indices pour rouge, vert, bleu, alpha galère à suivre
200 int dst_ir, dst_ig, dst_ib, dst_ia;
201 int src_ir, src_ig, src_ib, src_ia;
202 hw_surface_get_channel_indices(destination, &dst_ir, &dst_ig, &dst_ib, &dst_ia);
203 hw_surface_get_channel_indices(source, &src_ir, &src_ig, &src_ib, &src_ia);
204 // Bon, là on va copier ligne par ligne
205 for (int i = 0; i < dst_rect_reel.size.height; i++) {
206 // Pointeur vers le début de la ligne dans la surface destination
207 uint8_t* dst_pixel = pixel_0_dst + ((dst_rect_reel.top_left.y + i) * dst_surface_size.width + dst_rect_reel.top_left.x) * 4;
208 // Pointeur vers le début de la ligne dans la surface source
209 uint8_t* src_pixel = pixel_0_src + ((src_rect_reel.top_left.y + i) * src_surface_size.width + src_rect_reel.top_left.x) * 4;
210 // Si alpha est false, on copie tout direct
211 if (!alpha) {
212 for (int a = 0; a < dst_rect_reel.size.width; a++) {
213 // On copie les 4 canaux (RGBA) d’un coup, simple mais faut pas déborder
214 for (int b = 0; b < 4; ++b) {
215 dst_pixel[a * 4 + b] = src_pixel[a * 4 + b];
216 }
217 }
218 } else {
219 // Là c’est le mode alpha
220 for (int x = 0; x < dst_rect_reel.size.width; ++x) {
221 uint8_t* S_dst_pixel = dst_pixel + x * 4;
222 uint8_t* S_src_pixel = src_pixel + x * 4;
223 // On normalise l’alpha entre 0 et 1
224 float alpha_norma = (float)S_src_pixel[src_ia] / 255.0f;
225 // Mélange des couleurs avec la formule magique, merci le cours
226 S_dst_pixel[dst_ir] = (uint8_t)(alpha_norma * S_src_pixel[src_ir] + (1 - alpha_norma) * S_dst_pixel[dst_ir]);
227 S_dst_pixel[dst_ig] = (uint8_t)(alpha_norma * S_src_pixel[src_ig] + (1 - alpha_norma) * S_dst_pixel[dst_ig]);
228 S_dst_pixel[dst_ib] = (uint8_t)(alpha_norma * S_src_pixel[src_ib] + (1 - alpha_norma) * S_dst_pixel[dst_ib]);
229 }
230 }
231 }
232 // Tout s’est bien passé, on peut rentrer chez nous !
233 return 0;
234}
