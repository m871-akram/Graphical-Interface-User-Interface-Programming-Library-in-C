5// Dessine une ligne entre deux points avec l'algo de Bresenham (ça fait des lignes bien droites !)
6void draw_line(ei_surface_t surface, ei_point_t point_1, ei_point_t point_2, ei_color_t couleur, const ei_rect_t* clipper)
7{
8 // On récupère le buffer (l'endroit où on dessine) et la taille de la surface
9 uint8_t* pixel_0 = hw_surface_get_buffer(surface);
10 ei_size_t taille_surface = hw_surface_get_size(surface);
11
12 // On convertit la couleur en un format que la surface comprend (ça dépend si t'es sur Mac, Windows ou Linux)
13 #if defined(_APPLE_) || defined(_WIN32)
14 uint32_t valeur_pixel = ei_impl_map_rgba(surface, couleur);
15 #else
16 uint32_t valeur_pixel = *((uint32_t*)&couleur);
17 #endif
18
19 // On définit une zone où on a le droit de dessiner (le "clipper")
20 int clip_xmin = 0, clip_ymin = 0, clip_xmax = taille_surface.width - 1, clip_ymax = taille_surface.height - 1;
21 if (clipper != NULL) {
22 clip_xmin = clipper->top_left.x;
23 clip_ymin = clipper->top_left.y;
24 clip_xmax = clip_xmin + clipper->size.width - 1;
25 clip_ymax = clip_ymin + clipper->size.height - 1;
26 }
27
28 // On récupère les coordonnées des deux points
29 int x1 = point_1.x, y1 = point_1.y;
30 int x2 = point_2.x, y2 = point_2.y;
31
32 // On calcule les différences entre les points (pour savoir de combien on bouge)
33 int dx = abs(x2 - x1);
34 int dy = abs(y2 - y1);
35 int sx = (x2 >= x1) ? 1 : -1; // Direction : +1 si on va à droite, -1 à gauche
36 int sy = (y2 >= y1) ? 1 : -1; // Direction : +1 si on va en bas, -1 en haut
37
38 // On vérifie si la ligne est plus large que haute
39 bool dirige_par_x = dx >= dy;
40
41 // On commence au premier point
42 int x = x1, y = y1;
43
44 // On initialise une erreur pour l'algo de Bresenham (ça aide à choisir les bons pixels)
45 int E = dirige_par_x ? dy : dx;
46
47 // On calcule où est le premier pixel dans le buffer
48 uint8_t* pixel_ptr = pixel_0 + (y1 * taille_surface.width +x1 )* 4;
49 int ligne_increment = sy * taille_surface.width * 4; // Pour sauter à la ligne suivante
50 int pixel_increment = sx * 4; // Pour bouger d'un pixel
51
52 // Boucle pour dessiner la ligne pixel par pixel
53 while (true) {
54 // On dessine seulement si le pixel est dans la zone autorisée
55 if (x >= 0 && x < taille_surface.width && y >= 0 && y < taille_surface.height &&
56 x >= clip_xmin && x <= clip_xmax && y >= clip_ymin && y <= clip_ymax) {
57 *(uint32_t*)pixel_ptr = valeur_pixel;
58 }
59
60 // Si on est arrivé au point final, on arrête
61 if (x == x2 && y == y2) {
62 break;
63 }
64
65 // On bouge le pixel suivant selon l'axe dominant
66 if (dirige_par_x) {
67 // Si la ligne est plus horizontale
68 x += sx;
69 pixel_ptr += pixel_increment;
70 E += dy;
71 if (2 * E > dx) {
72 y += sy;
73 pixel_ptr += ligne_increment;
74 E -= dx;
75 }
76 } else {
77 // Si la ligne est plus verticale
78 y += sy;
79 pixel_ptr += ligne_increment;
80 E += dx;
81 if (2 * E > dy) {
82 x += sx;
83 pixel_ptr += pixel_increment;
84 E -= dy;
85 }
86 }
87 }
88}

91void draw_horizontal_line(ei_surface_t surface, int x1, int x2, int y, ei_color_t couleur, const ei_rect_t* clipper)
92{
93 // Si x1 est plus grand que x2, on les échange pour dessiner dans le bon sens
94 if (x1 > x2) {
95 int a = x1;
96 x1 = x2;
97 x2 = a;

102 ei_size_t taille_surface = hw_surface_get_size(surface);
103
104 // On convertit la couleur (pareil, ça dépend de la plateforme)
105 #if defined(_APPLE_) || defined(_WIN32)

108 uint32_t valeur_pixel = *((uint32_t*)&couleur);
109 #endif
110
111 // On définit la zone où on peut dessiner

113 if (clipper) {
114 clip_xmin = clipper->top_left.x;
115 clip_xmax = clipper->top_left.x + clipper->size.width - 1;

117 clip_ymax = clipper->top_left.y + clipper->size.height - 1;
118 }
119
120 // Si y est hors de la zone, on ne dessine rien
121 if (y < clip_ymin || y > clip_ymax || y < 0 || y >= taille_surface.height) return;
122
123 // On ajuste x1 et x2 pour qu'ils restent dans les limites
124 x1 = (x1 > clip_xmin) ? x1 : clip_xmin;
125 x1 = (x1 < 0) ? 0 : x1;
126 x2 = (x2 < clip_xmax) ? x2 : clip_xmax;
127 x2 = (x2 >= taille_surface.width) ? taille_surface.width - 1 : x2;
128
129 // Si x1 > x2 après ajustement, rien à dessiner
130 if (x1 > x2) return;
131
132 // On trouve le premier pixel à dessiner
133 uint32_t* ptr_pixel = (uint32_t*)(pixel_0 + (y * taille_surface.width + x1) * 4);
134 // On boucle pour colorier chaque pixel
135 for (int x = x1; x <= x2; x++) {
136 *ptr_pixel++ = valeur_pixel;
137 }
138}

140// Calcule l'inverse de la pente entre deux points (pour savoir comment une ligne bouge)
141float calcule_inverse_pente(ei_point_t p1, ei_point_t p2)
142{
143 // Si les y sont égaux, pas de pente, on renvoie 0
144 return ((p2.y - p1.y) != 0) ? (float)(p2.x - p1.x) / (p2.y - p1.y) : 0;

147// Crée une table avec les arêtes d’un polygone (pour dessiner des formes comme des octogones)
148void creer_table_tc(const ei_point_t* point_array, size_t point_array_size, edge_t* edge_table[], int y_min, int y_max)
149{
150 // On boucle sur tous les points du polygone
151 for (size_t i = 0; i < point_array_size; i++) {
152 ei_point_t p1 = point_array[i];
153 ei_point_t p2 = point_array[(i + 1) % point_array_size];
154
155 // Si les points sont à la même hauteur, on skip (pas d’arête utile)
156 if (p1.y == p2.y) continue;
157
158 // On met le point le plus bas en p_min et l’autre en p_max
159 ei_point_t p_min = (p1.y < p2.y) ? p1 : p2;
160 ei_point_t p_max = (p1.y < p2.y) ? p2 : p1;
161
162 // Si l’arête est hors de la zone qu’on veut, on skip
163 if (p_max.y <= y_min || p_min.y >= y_max) continue;
164
165 // On crée une nouvelle arête
166 edge_t* edge = (edge_t*)malloc(sizeof(edge_t));
167 if (!edge) return; // Si ça marche pas, on arrête
168
169 // On remplit les infos de l’arête
170 edge->ymax = p_max.y;
171 edge->inv_m = calcule_inverse_pente(p_min, p_max);
172
173 // Si le point est trop bas, on ajuste x pour commencer à y_min
174 if (p_min.y < y_min) {
175 edge->x = p_min.x + edge->inv_m * (y_min - p_min.y);
176 p_min.y = y_min;
177 } else {
178 edge->x = p_min.x;
179 }
180
181 // On calcule où mettre l’arête dans la table
182 int index = p_min.y - y_min;
183 if (index >= 0 && index < (y_max - y_min)) {
184 edge->next = edge_table[index];
185 edge_table[index] = edge; // On ajoute l’arête à la table
186 } else {
187 free(edge); // Si c’est hors limite, on libère
188 }
189 }
190}

200// Supprime les arêtes qui sont finies dans la liste TCA
201void supprimer_arete_tca(int y, edge_t** tca)
202{
203 edge_t* current = *tca;
204 edge_t* prev = NULL;
205
206 // On parcourt la liste
207 while (current != NULL) {
208 // Si l’arête se termine à cette ligne (y), on la vire
209 if (current->ymax == y) {
210 edge_t* u = current;
211 if (prev == NULL) { // Si c’est le début de la liste
212 *tca = current->next;
213 current = *tca;
214 } else {
215 prev->next = current->next;
216 current = prev->next;
217 }
218 free(u); // On libère la mémoire
219 } else {
220 prev = current;
221 current = current->next;
222 }
223 }

226// Trie la liste TCA pour mettre les arêtes dans l’ordre des x (de gauche à droite)
227void trier_tca_x(edge_t** tca)
228{
229 // Si la liste est vide ou a un seul élément, rien à faire
230 if (*tca == NULL || (*tca)->next == NULL) return;
231
232 edge_t* sorted = NULL;
233 edge_t* current = *tca;
234
235 // On parcourt chaque arête
236 while (current != NULL) {
237 edge_t* next = current->next;
238
239 // On insère l’arête dans la liste triée
240 if (sorted == NULL || current->x <= sorted->x) {
241 current->next = sorted;
242 sorted = current;
243 } else {
244 edge_t* u = sorted;
245 while (u->next != NULL && u->next->x < current->x) {
246 u = u->next;
247 }
248 current->next = u->next;
249 u->next = current;
250 }
251
252 current = next;
253 }
254
255 // On met la liste triée à la place de l’ancienne
256 *tca = sorted;
257}

275ei_rect_t intersection_rect(const ei_rect_t* dst_rect, const ei_rect_t* clipper, ei_rect_t* src_rect)
276{
277 ei_rect_t result_rect = *dst_rect; // Copie initiale pour ne pas modifier dst_rect
278
279 if (clipper != NULL) {
280 // Calculer les coordonnées de l'intersection
281 int max_x_deb = (dst_rect->top_left.x > clipper->top_left.x) ? dst_rect->top_left.x : clipper->top_left.x;
282 int max_y_deb = (dst_rect->top_left.y > clipper->top_left.y) ? dst_rect->top_left.y : clipper->top_left.y;
283
284 int min_x_fin = (dst_rect->top_left.x + dst_rect->size.width < clipper->top_left.x + clipper->size.width) ?
285 (dst_rect->top_left.x + dst_rect->size.width) : (clipper->top_left.x + clipper->size.width);
286 int min_y_fin = (dst_rect->top_left.y + dst_rect->size.height < clipper->top_left.y + clipper->size.height) ?
287 (dst_rect->top_left.y + dst_rect->size.height) : (clipper->top_left.y + clipper->size.height);
288
289 // Calculer les dimensions de l'intersection
290 result_rect.top_left.x = max_x_deb;
291 result_rect.top_left.y = max_y_deb;
292 result_rect.size.width = min_x_fin - max_x_deb;
293 result_rect.size.height = min_y_fin - max_y_deb;
294
295 // Si l'intersection est vide, retourner un rectangle vide
296 if (result_rect.size.width <= 0 || result_rect.size.height <= 0) {
297 result_rect.size.width = 0;
298 result_rect.size.height = 0;
299 return result_rect;
300 }
301
302 // Ajuster le rectangle source si nécessaire
303 if (src_rect != NULL) {
304 src_rect->top_left.x = result_rect.top_left.x - dst_rect->top_left.x;
305 src_rect->top_left.y = result_rect.top_left.y - dst_rect->top_left.y;
306 src_rect->size = result_rect.size;
307 }
308 }
309
310 return result_rect;
311}
