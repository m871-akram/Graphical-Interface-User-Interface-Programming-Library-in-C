27static ei_rect_t button_get_content_area(ei_widget_t widget) {
28 ei_rect_t content_area = widget->screen_location;
29 content_area.top_left.x += widget->border_width;
30 content_area.top_left.y += widget->border_width;
31 content_area.size.width -= 2 * widget->border_width;
32 content_area.size.height -= 2 * widget->border_width;
33 if (content_area.size.width < 0) content_area.size.width = 0;
34 if (content_area.size.height < 0) content_area.size.height = 0;
35 return content_area;
36}

143// Fonction pour vérifier si un point est dans un rectangle (si elle n'est pas déjà globale)
144// Si vous l'avez déjà définie ailleurs (par ex. toplevel.c), vous pouvez l'enlever d'ici
145// et la déclarer dans un .h commun ou la rendre static si elle est locale à un module.
146// Pour cet exemple, je la laisse ici pour que la fonction soit autonome.
147static bool point_in_rect(ei_point_t point, const ei_rect_t* rect) {
148 return (point.x >= rect->top_left.x &&
149 point.x < rect->top_left.x + rect->size.width &&
150 point.y >= rect->top_left.y &&
151 point.y < rect->top_left.y + rect->size.height);
152}
