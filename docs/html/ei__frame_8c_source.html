29// Fonction pour calculer la zone de contenu d'un widget
30static ei_rect_t get_content_area(ei_widget_t widget) {
31 ei_rect_t content_area = widget->screen_location;
32 content_area.top_left.x += widget->border_width;
33 content_area.top_left.y += widget->border_width;
34 content_area.size.width -= 2 * widget->border_width;
35 content_area.size.height -= 2 * widget->border_width;
36 if (content_area.size.width < 0) content_area.size.width = 0;
37 if (content_area.size.height < 0) content_area.size.height = 0;
38 return content_area;
39}

42void frame_drawfunc(ei_widget_t widget, ei_surface_t surface, ei_surface_t pick_surface, ei_rect_t* clipper) {
43 ei_impl_widget_t* frame = (ei_impl_widget_t*)widget;
44
45 // 1. Dessiner le fond et le relief du frame
46 // Utiliser draw_button de ei_relief.c avec un rayon de 0 pour les coins droits.
47 // L'état est toujours "raised" ou "sunken" ou "none", pas d'état "pressé" pour un frame.
48 if (frame->relief == ei_relief_none && frame->border_width == 0) {
49 ei_fill(surface, &(frame->bg_color), &(frame->screen_location)); // Simple remplissage
50 } else {
51 // La fonction draw_button s'attend à un rayon. Pour un cadre, le rayon est 0.
52 // Elle dessine aussi le fond.
53 draw_button(surface, &(frame->screen_location), 0.0f, frame->bg_color, frame->relief);
54 // Remarque : draw_button doit gérer le clipper si elle appelle ei_draw_polygon.
55 // Si draw_button ne prend pas de clipper, il faudrait clipper ici en amont/aval.
56 // On suppose que draw_button dans ei_relief.c utilise le clipper passé à ei_draw_polygon.
57 }
58
59 // 2. Dessiner le texte ou l'image
60 ei_rect_t content_area = get_content_area(widget);
61
62 if (frame->text != NULL && strlen(frame->text) > 0) {
63 ei_size_t text_size;
64 hw_text_compute_size(frame->text, frame->text_font, &text_size.width, &text_size.height);
65 ei_point_t text_pos;
66
67 // Calculer text_pos en fonction de frame->text_anchor, content_area et text_size
68 switch (frame->text_anchor) {
69 case ei_anc_northwest: text_pos = content_area.top_left; break;
70 case ei_anc_north: text_pos.x = content_area.top_left.x + (content_area.size.width - text_size.width) / 2; text_pos.y = content_area.top_left.y; break;
71 // ... autres ancres ...
72 case ei_anc_center:
73 default:
74 text_pos.x = content_area.top_left.x + (content_area.size.width - text_size.width) / 2;
75 text_pos.y = content_area.top_left.y + (content_area.size.height - text_size.height) / 2;
76 break;
77 }
78 ei_draw_text(surface, &text_pos, frame->text, frame->text_font, frame->text_color, &content_area);
79 } else if (frame->img != NULL) {
80 ei_rect_t src_img_rect;
81 if (frame->img_rect_ptr != NULL) {
82 src_img_rect = *(frame->img_rect_ptr);
83 } else {
84 src_img_rect.top_left = ei_point_zero();
85 src_img_rect.size = hw_surface_get_size(frame->img);
86 }
87
88 ei_size_t display_img_size = src_img_rect.size;
89 ei_point_t img_pos;
90
91 // Calculer img_pos en fonction de frame->img_anchor, content_area et display_img_size
92 switch (frame->img_anchor) {
93 case ei_anc_northwest: img_pos = content_area.top_left; break;
94 // ... autres ancres ...
95 case ei_anc_center:
96 default:
97 img_pos.x = content_area.top_left.x + (content_area.size.width - display_img_size.width) / 2;
98 img_pos.y = content_area.top_left.y + (content_area.size.height - display_img_size.height) / 2;
99 break;
100 }
101 ei_rect_t dst_img_rect = {img_pos, display_img_size};
102
103 // Il faut s'assurer que dst_img_rect est clippé par content_area avant d'appeler ei_copy_surface
104 // ou que ei_copy_surface gère le clipping par rapport à dst_img_rect dans la surface de destination.
105 // Pour l'instant, on suppose que ei_copy_surface le fait.
106 ei_copy_surface(surface, &dst_img_rect, frame->img, &src_img_rect, hw_surface_has_alpha(frame->img));
107 }
108 // 3. Dessiner les enfants
109 ei_rect_t children_clipper_final;
110
111 // Premier argument de intersection_rect est le rectangle à clipper (content_area)
112 // Deuxième argument est le clipper externe.
113 // Troisième argument (src_rect) n'est pas utile ici, donc on passe NULL.
114 children_clipper_final = intersection_rect(&content_area, clipper, NULL);
115 // Note: 'clipper' est déjà un ei_rect_t*
116 // 'content_area' est une variable locale ei_rect_t, donc on passe &content_area
117
118 if (children_clipper_final.size.width > 0 && children_clipper_final.size.height > 0) {
119 ei_impl_widget_draw_children(widget, surface, pick_surface, &children_clipper_final);
120 }
121
122 // 4. Dessiner sur le pick_surface pour le frame lui-même
123 if (pick_surface) {
124 ei_fill(pick_surface, &(frame->pick_color), &(frame->screen_location));
125 }
126
127}

154void ei_frame_configure(ei_widget_t widget,
155 ei_size_t* requested_size,
156 const ei_color_t* color,
157 int* border_width,
158 ei_relief_t* relief,
159 ei_string_t* text,
160 ei_font_t* text_font,
161 ei_color_t* text_color,
162 ei_anchor_t* text_anchor,
163 ei_surface_t* img,
164 ei_rect_ptr_t* img_rect,
165 ei_anchor_t* img_anchor) {
166
167 ei_impl_widget_t* frame = (ei_impl_widget_t*)widget;
168
169 if (requested_size) frame->requested_size = *requested_size;
170 if (color) frame->bg_color = *color;
171 if (border_width) frame->border_width = *border_width;
172 if (relief) frame->relief = *relief;
173
174 if (text) {
175 if (frame->text) free(frame->text);
176 if (*text) frame->text = strdup(*text); else frame->text = NULL;
177 }
178 if (text_font) frame->text_font = *text_font;
179 if (text_color) frame->text_color = *text_color;
180 if (text_anchor) frame->text_anchor = *text_anchor;
181
182 if (img) frame->img = *img; // Référence
183 if (img_rect) {
184 if (*img_rect == NULL) {
185 frame->img_rect_ptr = NULL;
186 } else {
187 frame->img_rect_val = **img_rect; // Copie de la valeur
188 frame->img_rect_ptr = &(frame->img_rect_val);
189 }
190 }
191 if (img_anchor) frame->img_anchor = *img_anchor;
192
193 // Si la taille ou le texte/image change, il faut potentiellement recalculer la géométrie
194 if (requested_size || text || img) {
195 if (widget->placer_params) { // Si déjà placé
196 // ei_app_invalidate_rect(&(widget->screen_location)); // Invalider l'ancienne
197 ei_impl_placer_run(widget); // Recalculer et invalider la nouvelle
198 }
199 } else if (color || border_width || relief) {
200 // Juste un redessin est nécessaire si la géométrie ne change pas
201 ei_app_invalidate_rect(&(widget->screen_location));
202 }
