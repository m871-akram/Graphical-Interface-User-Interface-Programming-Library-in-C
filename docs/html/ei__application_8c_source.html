85static void redraw_invalidated_areas() {
86 if (g_root_widget == NULL || g_root_widget->wclass == NULL || g_root_widget->wclass->drawfunc == NULL) {
87 return;
88 }
89 if (g_invalidated_rects_head == NULL) {
90 return;
91 }
92
93 hw_surface_lock(g_root_surface);
94 hw_surface_lock(g_pick_surface);
95
96 ei_color_t pick_clear_color = {0, 0, 0, 0xff};
97 ei_fill(g_pick_surface, &pick_clear_color, NULL);
98
99 // Parcourir chaque rectangle invalidé et redessiner cette zone
100 ei_linked_rect_t* current_invalid_rect_node = g_invalidated_rects_head;
101 while (current_invalid_rect_node != NULL) {
102 // Le clipper pour la drawfunc du root_widget est le rectangle invalidé courant
103 g_root_widget->wclass->drawfunc(g_root_widget, g_root_surface, g_pick_surface, &(current_invalid_rect_node->rect));
104 current_invalid_rect_node = current_invalid_rect_node->next;
105 }
106 // NE PAS LIBÉRER g_invalidated_rects_head ICI
107
108 hw_surface_unlock(g_pick_surface);
109 hw_surface_unlock(g_root_surface);
110
111 // Mettre à jour uniquement les zones qui ont été invalidées
112 hw_surface_update_rects(g_root_surface, g_invalidated_rects_head); // Utilise la liste
113
114 // Vider la liste des rectangles invalidés APRÈS hw_surface_update_rects
115 ei_linked_rect_t* node_to_free = g_invalidated_rects_head;
116 while (node_to_free != NULL) {
117 ei_linked_rect_t* next_node = node_to_free->next;
118 free(node_to_free);
119 node_to_free = next_node;
120 }
121 g_invalidated_rects_head = NULL;
122} // Fin de redraw_invalidated_areas

125void ei_app_run(void) {
126 ei_event_t event;
127 g_application_quit_request = false;
128
129 if (g_root_widget == NULL) {
130 fprintf(stderr, "ei_app_run: Erreur critique - Widget racine non initialisé.\n");
131 return;
132 }
133
134 ei_app_invalidate_rect(&(g_root_widget->screen_location));
135 redraw_invalidated_areas();
136
137 while (!g_application_quit_request) {
138 hw_event_wait_next(&event);
139
140 ei_widget_t target_widget = NULL;
141 bool event_handled_by_target = false;
142
143 ei_widget_t current_active_widget = ei_event_get_active_widget();
144 ei_widget_t current_focused_widget = ei_app_get_focus();
145
146 if (current_active_widget != NULL) {
147 target_widget = current_active_widget;
148 } else if ((event.type == ei_ev_keydown || event.type == ei_ev_keyup || event.type == ei_ev_text_input) && current_focused_widget != NULL) {
149 target_widget = current_focused_widget;
150 } else if (event.type >= ei_ev_mouse_buttondown && event.type <= ei_ev_mouse_wheel) {
151 target_widget = ei_widget_pick(&(event.param.mouse.where));
152 // Optionnel: Donner le focus au widget cliqué s'il est "focusable"
153 // if (target_widget != NULL && target_widget != ei_app_root_widget()) {
154 // // Ajouter une vérification si la classe du widget est focusable
155 // // par exemple, if (target_widget->wclass->is_focusable)
156 // ei_app_set_focus(target_widget);
157 // }
158 }
159
160 if (target_widget != NULL && target_widget->wclass != NULL && target_widget->wclass->handlefunc != NULL) {
161 event_handled_by_target = target_widget->wclass->handlefunc(target_widget, &event);
162 }
163
164 if (!event_handled_by_target) {
165 ei_default_handle_func_t dhf = ei_event_get_default_handle_func();
166 if (dhf != NULL) {
167 dhf(&event);
168 }
169 }
170
171 if (!g_application_quit_request) {
172 // Le redessin se fait s'il y a des rectangles invalidés.
173 // ei_app_invalidate_rect a dû être appelé par un handlefunc ou autre pour
174 // qu'un redessin ait lieu.
175 // Si vous voulez forcer un redessin après chaque événement pour le moment (debug):
176 // ei_app_invalidate_rect(&(g_root_widget->screen_location));
177 redraw_invalidated_areas(); // Ne fait rien si g_invalidated_rects_head est NULL
178 }
179 }

182void ei_app_free(void) {
183 ei_linked_rect_t* current_invalid = g_invalidated_rects_head;
184 while (current_invalid != NULL) {
185 ei_linked_rect_t* next = current_invalid->next;
186 free(current_invalid);
187 current_invalid = next;

189 g_invalidated_rects_head = NULL;
190
191 if (g_root_widget != NULL) {
192 ei_widget_destroy(g_root_widget);
193 g_root_widget = NULL;
194 }
195 if (g_pick_surface != NULL) {
196 hw_surface_free(g_pick_surface);
197 g_pick_surface = NULL;

199 if (g_root_surface != NULL) {
200 hw_surface_free(g_root_surface);
201 g_root_surface = NULL;
202 }

204 hw_text_font_free(ei_default_font);
205 ei_default_font = NULL;
206 }
207 hw_quit();
208}

218void ei_app_invalidate_rect(const ei_rect_t* rect) {
219 if (rect == NULL || rect->size.width <= 0 || rect->size.height <= 0) {
220 return;
221 }
222
223 ei_linked_rect_t* new_invalid_node = malloc(sizeof(ei_linked_rect_t));
