9const ei_rect_t* ei_widget_get_content_rect(ei_widget_t widget) {
10 if (widget == NULL) {
11 // Gérer le cas où le widget est NULL, peut-être retourner NULL
12 // ou un rectangle statique vide pour éviter les crashs, mais c'est un cas d'erreur.
13 static ei_rect_t null_rect = {{0,0},{0,0}}; // Pour l'exemple
14 return &null_rect; // Ou return NULL; et le code appelant doit vérifier
15 }
16
17 // Supposons que ei_impl_widget_t a un champ ei_rect_t* content_rect
18 // qui pointe soit vers screen_location, soit vers un rectangle calculé
19 // (par exemple, screen_location moins les bordures).
20 if (widget->content_rect != NULL) {
21 return widget->content_rect;
22 } else {
23 // Cas par défaut ou d'erreur : si content_rect n'est pas initialisé,
24 // on pourrait retourner screen_location comme fallback, mais c'est un signe
25 // que l'initialisation ou la mise à jour du content_rect n'a pas été faite.
26 // Pour un frame simple, content_rect pourrait pointer sur screen_location
27 // ou être calculé pour exclure les bordures.
28 // Par défaut, on a mis widget->content_rect = &(widget->screen_location);
29 // dans ei_widget_create, donc ce cas ne devrait pas arriver si widget est valide.
30 // Si content_rect peut être NULL pour une raison, retourner screen_location
31 // peut être un comportement par défaut raisonnable.
32 return &(widget->screen_location);
33 }
34}

44const ei_color_t* ei_widget_get_pick_color(ei_widget_t widget) {
45 if (!widget) {
46 // Optionnel : retourner une couleur par défaut ou NULL si le widget est NULL.
47 // Le sujet ne spécifie pas, mais retourner une couleur valide peut éviter des crashs.
48 // Cependant, si le widget est NULL, l'appelant devrait le savoir.
49 // Pour être sûr, on peut retourner une couleur noire transparente ou NULL.
50 // Pour la simplicité, on peut retourner NULL et l'appelant doit vérifier.
51 // Mais le code du test mouse_tracker fait :
52 // const ei_color_t* p_color = ei_widget_get_pick_color(picked_widget);
53 // if (p_color) { g_picked_widget_color = *p_color; }
54 // Donc retourner NULL est géré.
55 // Cependant, si widget est NULL, widget->pick_color n'est pas accessible.
56
57 // Mieux : si widget est NULL, on retourne une couleur "nulle" statique
58 // pour éviter de déréférencer un pointeur NULL plus tard si l'appelant ne vérifie pas.
59 static const ei_color_t null_pick_color = {0, 0, 0, 0}; // Noir complètement transparent
60 return &null_pick_color;
61 // Ou simplement : return NULL; (et l'appelant *doit* vérifier)
62 }
63 // Assumant que ei_impl_widget_t a un champ pick_color
64 return &(widget->pick_color);
65}

100 widget->requested_size = requested_size;
101
102 // Si la taille demandée change ET que le widget est actuellement géré par un placeur,
