79static void toplevel_update_content_rect(ei_widget_t widget) {
80 ei_impl_toplevel_t* toplevel = (ei_impl_toplevel_t*)widget;
81 ei_impl_widget_t* w = &(toplevel->widget);

83 // Si content_rect pointe par défaut vers screen_location, on doit le recalculer.
84 // Idéalement, ei_impl_widget_t aurait un ei_rect_t content_rect_value;
85 // et content_rect serait un pointeur dessus.
86 // Pour cet exemple, on suppose qu'on peut modifier w->content_rect.
87 // IMPORTANT: Si widget->content_rect pointe sur widget->screen_location,
88 // il faut allouer dynamiquement un nouveau rectangle pour content_rect ou utiliser une valeur dédiée.
89 // Pour cet exemple, on va tricher un peu et supposer que l'appelant

91 // La solution propre est que ei_widget_set_content_rect soit appelée par geomnotify.

94 content_r.top_left.x = w->screen_location.top_left.x + w->border_width;

96 content_r.size.width = w->screen_location.size.width - 2 * w->border_width;
97 content_r.size.height = w->screen_location.size.height - 2 * w->border_width - TOPLEVEL_TITLE_BAR_HEIGHT;

100 if (content_r.size.height < 0) content_r.size.height = 0;

102 // Ceci est pour que ei_widget_get_content_rect() retourne la bonne chose.
103 // Si widget->content_rect pointe sur widget->screen_location, il FAUT le changer.
104 // Si vous avez une structure dédiée pour widget->content_rect_value :
105 // widget->content_rect_value = content_r;
106 // widget->content_rect = &widget->content_rect_value;
107 // Pour l'instant, on ne le fait pas pour ne pas casser la structure ei_impl_widget_t.
108 // La fonction draw_children doit être intelligente.

111 if (w->content_rect == &(w->screen_location)) {
112 // Ne pas faire ça ici si on ne veut pas allouer dynamiquement le content_rect.
113 // La drawfunc et get_content_rect de la classe toplevel s'en chargeront.

136void toplevel_drawfunc(ei_widget_t widget, ei_surface_t surface, ei_surface_t pick_surface, ei_rect_t* clipper) {
137 ei_impl_toplevel_t* toplevel = (ei_impl_toplevel_t*)widget;
138 ei_impl_widget_t* w = &(toplevel->widget);
139
140 // 1. Dessiner le fond du toplevel (la zone de contenu)
141 // On calcule la zone de contenu interne (après la barre de titre et les bordures)
142 ei_rect_t content_bg_rect;
143 content_bg_rect.top_left.x = w->screen_location.top_left.x + w->border_width;
144 content_bg_rect.top_left.y = w->screen_location.top_left.y + w->border_width + TOPLEVEL_TITLE_BAR_HEIGHT;
145 content_bg_rect.size.width = w->screen_location.size.width - 2 * w->border_width;
146 content_bg_rect.size.height = w->screen_location.size.height - 2 * w->border_width - TOPLEVEL_TITLE_BAR_HEIGHT;
147 if (content_bg_rect.size.width < 0) content_bg_rect.size.width = 0;
148 if (content_bg_rect.size.height < 0) content_bg_rect.size.height = 0;
149
150 ei_fill(surface, &(w->bg_color), &content_bg_rect);
151
152
153 // 2. Dessiner la bordure du toplevel (autour de tout)
154 // On pourrait utiliser draw_button avec rayon 0, mais ici on fait une bordure simple
155 if (w->border_width > 0) {
156 ei_color_t border_color = {0x30, 0x30, 0x30, 0xff}; // Gris foncé pour la bordure
157 // Dessin des 4 rectangles de bordure
158 ei_rect_t border_rect;
159 // Haut
160 border_rect.top_left = w->screen_location.top_left;
161 border_rect.size = ei_size(w->screen_location.size.width, w->border_width);
162 ei_fill(surface, &border_color, &border_rect);
163 // Bas
164 border_rect.top_left.y = w->screen_location.top_left.y + w->screen_location.size.height - w->border_width;
165 ei_fill(surface, &border_color, &border_rect);
166 // Gauche
167 border_rect.top_left.y = w->screen_location.top_left.y + w->border_width; // Pour ne pas redessiner les coins
168 border_rect.size = ei_size(w->border_width, w->screen_location.size.height - 2 * w->border_width);
169 ei_fill(surface, &border_color, &border_rect);
170 // Droite
171 border_rect.top_left.x = w->screen_location.top_left.x + w->screen_location.size.width - w->border_width;
172 ei_fill(surface, &border_color, &border_rect);
173 }
174
175 // 3. Dessiner la barre de titre
176 ei_rect_t title_bar_rect = get_title_bar_rect(widget);
177 ei_color_t title_bar_color = {0x80, 0x80, 0x80, 0xff}; // Gris moyen
178 ei_fill(surface, &title_bar_color, &title_bar_rect);
179
180 // 4. Dessiner le titre
181 if (toplevel->title) {
182 ei_size_t text_size;
183 hw_text_compute_size(toplevel->title, w->text_font, &text_size.width, &text_size.height);
184 ei_point_t text_pos;
185 text_pos.x = title_bar_rect.top_left.x + 5; // Marge
186 if (toplevel->closable) text_pos.x += TOPLEVEL_DECORATION_SIZE + 2; // Espace pour le bouton
187 text_pos.y = title_bar_rect.top_left.y + (title_bar_rect.size.height - text_size.height) / 2;
188 ei_draw_text(surface, &text_pos, toplevel->title, w->text_font, w->text_color, &title_bar_rect);
189 }
190
191 // 5. Dessiner le bouton de fermeture
192 if (toplevel->closable) {
193 ei_rect_t close_rect = get_close_button_rect(widget);
194 ei_color_t close_color = {0xff, 0x60, 0x60, 0xff}; // Rougeâtre
195 ei_fill(surface, &close_color, &close_rect);
196 // Dessiner un 'X'
197 ei_point_t p1 = {close_rect.top_left.x + 3, close_rect.top_left.y + 3};
198 ei_point_t p2 = {close_rect.top_left.x + close_rect.size.width - 4, close_rect.top_left.y + close_rect.size.height - 4};
199 ei_point_t p3 = {close_rect.top_left.x + 3, close_rect.top_left.y + close_rect.size.height - 4};
200 ei_point_t p4 = {close_rect.top_left.x + close_rect.size.width - 4, close_rect.top_left.y + 3};
201 ei_color_t x_color = {0x00, 0x00, 0x00, 0xff};
202 ei_point_t lines_x[] = {p1, p2, p3, p4};
203 ei_draw_polyline(surface, &lines_x[0], 2, x_color, &close_rect);
204 ei_draw_polyline(surface, &lines_x[2], 2, x_color, &close_rect);
205 }
206
207 // 6. Dessiner la poignée de redimensionnement
208 if (toplevel->resizable != ei_axis_none) {
209 ei_rect_t resize_rect = get_resize_handle_rect(widget);
210 ei_color_t resize_color = {0x60, 0x60, 0xff, 0xff}; // Bleuâtre
211 ei_fill(surface, &resize_color, &resize_rect);
212 }
213
214 // 7. Dessiner les enfants
215 ei_rect_t children_clipper_final; // Ce sera le clipper effectif pour les enfants
216
217 if (clipper != NULL) {
218 // Calculer l'intersection de content_bg_rect (la zone où les enfants peuvent être)
219 // et du clipper externe passé à toplevel_drawfunc.
220 // Le résultat est le nouveau clipper pour les enfants.
221 children_clipper_final = intersection_rect(&content_bg_rect, clipper, NULL);
222 // Note : content_bg_rect est une variable locale de type ei_rect_t, donc on passe &content_bg_rect.
223 // clipper est un paramètre de type ei_rect_t*, on le passe directement.
224 } else {
225 // S'il n'y a pas de clipper externe, le clipper pour les enfants est simplement
226 // la zone de contenu du toplevel.
227 children_clipper_final = content_bg_rect;
228 }
229
230 // On dessine les enfants seulement si le clipper résultant a une taille positive
231 if (children_clipper_final.size.width > 0 && children_clipper_final.size.height > 0) {
232 ei_impl_widget_draw_children(widget, surface, pick_surface, &children_clipper_final);
233 }
234
235 // 8. Dessiner sur le pick_surface (uniquement le toplevel lui-même)
236 if (pick_surface) {
237 // w est (ei_impl_widget_t*)widget
238 ei_fill(pick_surface, &(w->pick_color), &(w->screen_location));
239 }
240}

268bool toplevel_handlefunc(ei_widget_t widget, struct ei_event_t* event) {
269 ei_impl_toplevel_t* toplevel = (ei_impl_toplevel_t*)widget;
270 ei_impl_widget_t* w = &(toplevel->widget);
271 bool event_handled = false;
272
273 switch (event->type) {
274 case ei_ev_mouse_buttondown:
275 if (event->param.mouse.button == ei_mouse_button_left) {
276 ei_point_t click_pos = event->param.mouse.where;
277 ei_rect_t title_bar_rect = get_title_bar_rect(widget);
278 ei_rect_t close_button_rect = get_close_button_rect(widget);
279 ei_rect_t resize_handle_rect = get_resize_handle_rect(widget);
280
281 if (toplevel->closable && point_in_rect(click_pos, &close_button_rect)) {
282 // Action de fermeture
283 if (w->destructor) w->destructor(widget); // Notifier l'utilisateur avant de détruire
284 ei_widget_destroy(widget); // Détruire la fenêtre
285 // Si c'est la dernière toplevel et qu'il n'y a plus que le root, quitter l'app?
286 // Cette logique est généralement gérée par l'application utilisateur.
287 event_handled = true;
288 } else if (toplevel->resizable != ei_axis_none && point_in_rect(click_pos, &resize_handle_rect)) {
289 // Commencer le redimensionnement
290 toplevel->is_resizing = true;
291 toplevel->drag_start_pos = click_pos;
292 toplevel->widget_start_size_on_drag = w->requested_size; // On redimensionne la taille de contenu demandée
293 ei_event_set_active_widget(widget);
294 event_handled = true;
295 } else if (point_in_rect(click_pos, &title_bar_rect)) {
296 // Commencer le déplacement
297 toplevel->is_moving = true;
298 toplevel->drag_start_pos = click_pos;
299 // La position est gérée par le placeur, on a besoin de la position de screen_location
300 if (w->placer_params) { // Devrait toujours être le cas si visible
301 toplevel->widget_start_pos_on_drag = ei_point(w->placer_params->x, w->placer_params->y);
302 } else { // Fallback (ne devrait pas arriver pour un toplevel placé)
303 toplevel->widget_start_pos_on_drag = w->screen_location.top_left;
304 // Ajuster pour être relatif au parent si le placeur utilise des relatifs
305 if (w->parent && w->parent->content_rect) {
306 toplevel->widget_start_pos_on_drag.x -= w->parent->content_rect->top_left.x;
307 toplevel->widget_start_pos_on_drag.y -= w->parent->content_rect->top_left.y;
308 }
309 }
310 ei_event_set_active_widget(widget);
311 event_handled = true;
312 }
313 // Si clic dans la zone de contenu, ne pas gérer ici (laisser les enfants ou le défaut)
314 }
315 break;
316
317 case ei_ev_mouse_move:
318 if (ei_event_get_active_widget() == widget) {
319 if (toplevel->is_moving) {
320 int dx = event->param.mouse.where.x - toplevel->drag_start_pos.x;
321 int dy = event->param.mouse.where.y - toplevel->drag_start_pos.y;
322 int new_x = toplevel->widget_start_pos_on_drag.x + dx;
323 int new_y = toplevel->widget_start_pos_on_drag.y + dy;
324 // Mettre à jour les paramètres du placeur et relancer le placement
325 if (w->placer_params) { // On modifie les x,y absolus du placeur
326 w->placer_params->x = new_x;
327 w->placer_params->y = new_y;
328 // ei_app_invalidate_rect(&(w->screen_location)); // Invalider l'ancienne position
329 ei_impl_placer_run(widget); // Recalcule la géométrie et invalide la nouvelle
330 }
331 event_handled = true;
332 } else if (toplevel->is_resizing) {
333 int dw = event->param.mouse.where.x - toplevel->drag_start_pos.x;
334 int dh = event->param.mouse.where.y - toplevel->drag_start_pos.y;
335
336 ei_size_t new_content_size = toplevel->widget_start_size_on_drag;
337 if (toplevel->resizable == ei_axis_x || toplevel->resizable == ei_axis_both) {
338 new_content_size.width += dw;
339 }
340 if (toplevel->resizable == ei_axis_y || toplevel->resizable == ei_axis_both) {
341 new_content_size.height += dh;
342 }
343
344 // Appliquer min_size
345 if (new_content_size.width < toplevel->min_size.width) new_content_size.width = toplevel->min_size.width;
346 if (new_content_size.height < toplevel->min_size.height) new_content_size.height = toplevel->min_size.height;
347
348 // Mettre à jour la requested_size du widget (qui est la taille de contenu pour toplevel)
349 // Et appeler ei_toplevel_configure pour que cela soit pris en compte.
350 // Cela va déclencher un ei_impl_placer_run si le placer est bien fait.
351 // ei_app_invalidate_rect(&(w->screen_location)); // Invalider l'ancienne
352 ei_toplevel_configure(widget, &new_content_size, NULL, NULL, NULL, NULL, NULL, NULL);
353 // ei_impl_placer_run(widget); // Est appelé par ei_toplevel_configure si la taille change
354 event_handled = true;
355 }
356 }
357 break;
358
359 case ei_ev_mouse_buttonup:
360 if (ei_event_get_active_widget() == widget) {
361 if (toplevel->is_moving || toplevel->is_resizing) {
362 toplevel->is_moving = false;
363 toplevel->is_resizing = false;
364 ei_event_set_active_widget(NULL);
365 // Pas besoin d'invalider ici, car move/resize le font déjà
366 event_handled = true;
367 }
368 }
369 break;
370 default:
371 break;
372 }
373 return event_handled;
374}

393void ei_toplevel_configure(ei_widget_t widget,
394 ei_size_t* requested_content_size, // Note: c'est la taille du *contenu*
395 const ei_color_t* color,
396 int* border_width,
397 ei_string_t* title,
398 bool* closable,
399 ei_axis_set_t* resizable,
400 ei_size_ptr_t* min_size_ptr) {
401
402 ei_impl_toplevel_t* toplevel = (ei_impl_toplevel_t*)widget;
403 ei_impl_widget_t* w = &(toplevel->widget);
404 bool geometry_changed = false;
405
406 if (requested_content_size) {
407 // La requested_size du widget toplevel dans ei_impl_widget_t
408 // est la taille de son *contenu*. Le placeur, lui, a besoin de la taille totale.
409 // Le placeur pour un toplevel devra donc calculer la taille totale
410 // (requested_content_size + décorations) s'il n'a que requested_content_size.
411 // Ou alors, ei_toplevel_configure met à jour w->requested_size pour que
412 // ce soit la taille totale, et stocke requested_content_size ailleurs.
413 // Pour l'API, requested_size de ei_frame_configure est la taille totale.
414 // Pour ei_toplevel_configure, c'est la taille du contenu.
415 // Solution : on stocke requested_content_size dans w->requested_size
416 // et le placer pour le toplevel l'interprétera comme taille de contenu.
417 if (w->requested_size.width != requested_content_size->width || w->requested_size.height != requested_content_size->height) {
418 w->requested_size = *requested_content_size;
419 geometry_changed = true;
420 }
421 }
422 if (color) w->bg_color = *color; // Couleur de fond du contenu
423 if (border_width) {
424 if (w->border_width != *border_width) {
425 w->border_width = *border_width;
426 geometry_changed = true; // Changement de bordure affecte la taille du contenu
427 }
428 }
429 if (title) {
430 if (toplevel->title) free(toplevel->title);
431 if (*title) toplevel->title = strdup(*title); else toplevel->title = NULL;
432 // Pas de changement de géométrie direct, mais redessin nécessaire
433 }
434 if (closable) toplevel->closable = *closable;
435 if (resizable) toplevel->resizable = *resizable;
436 if (min_size_ptr) {
437 if (*min_size_ptr == NULL) { // Demande de réinitialiser à la valeur par défaut
438 toplevel->min_size = ei_size(160, 120);
439 } else {
440 toplevel->min_size = **min_size_ptr;
441 }
442 // S'assurer que la taille actuelle respecte la nouvelle min_size
443 if (w->requested_size.width < toplevel->min_size.width) {
444 w->requested_size.width = toplevel->min_size.width; geometry_changed = true;
445 }
446 if (w->requested_size.height < toplevel->min_size.height) {
