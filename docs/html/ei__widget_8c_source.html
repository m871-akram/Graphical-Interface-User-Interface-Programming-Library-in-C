130void ei_impl_widget_draw_children(ei_widget_t widget,
131 ei_surface_t surface,
132 ei_surface_t pick_surface,
133 ei_rect_t* clipper) {
134 if (widget == NULL) return;
135
136 ei_widget_t current_child = widget->children_head;
137 ei_rect_t child_clipper; // Le clipper pour les enfants
138
139 // Le clipper pour les enfants est l'intersection du clipper actuel
140 // et du content_rect du widget parent.
141 // Le content_rect est déjà exprimé en coordonnées globales (écran).
142 const ei_rect_t* parent_content_rect = ei_widget_get_content_rect(widget);
143
144 if (clipper == NULL) { // Si pas de clipper externe, on utilise le content_rect du parent
145 child_clipper = *parent_content_rect;
146 } else {
147 // Calcul de l'intersection entre clipper et parent_content_rect
148 child_clipper.top_left.x = (clipper->top_left.x > parent_content_rect->top_left.x) ? clipper->top_left.x : parent_content_rect->top_left.x;
149 child_clipper.top_left.y = (clipper->top_left.y > parent_content_rect->top_left.y) ? clipper->top_left.y : parent_content_rect->top_left.y;
150
151 int clipper_right = clipper->top_left.x + clipper->size.width;
152 int parent_content_right = parent_content_rect->top_left.x + parent_content_rect->size.width;
153 child_clipper.size.width = ((clipper_right < parent_content_right) ? clipper_right : parent_content_right) - child_clipper.top_left.x;
154
155 int clipper_bottom = clipper->top_left.y + clipper->size.height;
156 int parent_content_bottom = parent_content_rect->top_left.y + parent_content_rect->size.height;
157 child_clipper.size.height = ((clipper_bottom < parent_content_bottom) ? clipper_bottom : parent_content_bottom) - child_clipper.top_left.y;
158
159 if (child_clipper.size.width <= 0 || child_clipper.size.height <= 0) {
160 return; // Le clipper résultant est vide, rien à dessiner pour les enfants.
161 }
162 }
163
164
165 while (current_child != NULL) {
166 if (current_child->placer_params != NULL) { // Seulement si le widget est géré par un placeur (donc visible)
167 if (current_child->wclass != NULL && current_child->wclass->drawfunc != NULL) {
168 current_child->wclass->drawfunc(current_child, surface, pick_surface, &child_clipper);
169 }
170 }
171 current_child = current_child->next_sibling;
172 }
173}

206ei_widget_t ei_widget_pick(ei_point_t* where) {
207 if (where == NULL || g_pick_surface == NULL) {
208 return NULL;
209 }
210
211 ei_size_t pick_surface_size = hw_surface_get_size(g_pick_surface);
212 if (where->x < 0 || where->x >= pick_surface_size.width ||
213 where->y < 0 || where->y >= pick_surface_size.height) {
214 return ei_app_root_widget(); // Clic en dehors, on considère que c'est le root (ou NULL si on préfère)
215 }
216
217 hw_surface_lock(g_pick_surface);
218 uint8_t* pixel_buffer = hw_surface_get_buffer(g_pick_surface);
219 int bytes_per_pixel = 4;
220 uint8_t* target_pixel_ptr = pixel_buffer + (where->y * pick_surface_size.width + where->x) * bytes_per_pixel;
221
222 int ir, ig, ib, ia;
223 hw_surface_get_channel_indices(g_pick_surface, &ir, &ig, &ib, &ia);
224
225 uint8_t r_val = target_pixel_ptr[ir];
226 uint8_t g_val = target_pixel_ptr[ig];
227 uint8_t b_val = target_pixel_ptr[ib];
228 // uint8_t a_val = (ia != -1) ? target_pixel_ptr[ia] : 0xff; // Alpha de la pick_color (normalement 0xff)
229
230 hw_surface_unlock(g_pick_surface);
231
232 // Reconstruire le pick_id
233 uint32_t picked_id = ((uint32_t)r_val << 16) | ((uint32_t)g_val << 8) | (uint32_t)b_val;
234
235 if (picked_id == 0) { // ID 0 est réservé pour le fond (non-widget cliquable)
236 return ei_app_root_widget();
237 }
238
239 ei_widget_t picked_widget = find_widget_by_pick_id(ei_app_root_widget(), picked_id);
240 return (picked_widget != NULL) ? picked_widget : ei_app_root_widget(); // Si non trouvé (ne devrait pas arriver si ID != 0), retourne root
241}
